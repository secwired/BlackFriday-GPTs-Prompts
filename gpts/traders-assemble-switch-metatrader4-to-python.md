    def __init__(self):

        self.addminperiod(self.params.period)



    def next(self):

        weight = 2.0 / (1.0 + self.params.period)



        if len(self.data) == self.params.period:

            last_ema = sum(self.data.get(size=self.params.period)) / self.params.period
